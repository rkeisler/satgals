SELECT 
   s.z as redshift,
   p.u, p.g, p.r, p.i, p.z,
   p.expAB_r, p.petroR50_r, p.petroR90_r
FROM PhotoObj AS p
   JOIN SpecObj AS s ON s.bestobjid = p.objid
WHERE
   s.z <= 0.01
   AND s.z > 0.005
   AND s.zWarning = 0
   AND p.rerun = 301
   AND type = 3
